Table: customer_customer_demo
Description:
  Junction table เชื่อมลูกค้า (customers) กับประเภทลูกค้า (customer_demographics) ใช้เมื่อต้องการรู้ว่าลูกค้าอยู่ใน demographic segment ไหนบ้าง
  ส่วนใหญ่ JOIN พร้อมกับ customers และ customer_demographics เสมอ
Columns:
  customer_id      VARCHAR(5) NOT NULL  -- FK → customers.customer_id
  customer_type_id VARCHAR(5) NOT NULL  -- FK → customer_demographics.customer_type_id
  PRIMARY KEY (customer_id, customer_type_id)
Relationships:
  - customer_id      → customers.customer_id             (many-to-one)
  - customer_type_id → customer_demographics.customer_type_id (many-to-one)
